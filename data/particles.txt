
#particle system name (referenced in-game)
sample {
    #system properties
    movieclip TestParticle         #particle movieclip linkage name. frame 1 = alive particle, frame 2 = "on death" animation
    max_particles 250      #max particles on screen at once
    layer 1                #0 = behind background tiles, 1 = behind player, 2 = in front, 3 = in front of tiles, 4 = in front of water, 5 = upper distortion layer, 6 = lower distortion layer, 7 = blurred foreground layer
    livebounds [-10, 64, -10, 42] #bounding box coords for when particles get destroyed off-screen. should not need to change. minx, maxx, miny, maxy
    prewarm [0,0] #for global particles, first value is number of iterations to prewarm, second value is time scale for each step
    batched false #delays the particles rendering by a frame (or more). use on systems with lots of particles for a performance boost if the frame delay is okay (should be fine for anything that doesnt interact with the player, like global particle effects)
    
    #emitter properties
    emit_rate [30,30]            #how often to emit particles. 10 = 10 per second, [5, 10] = random between 5 and 10 per second. set negative to make it a 1-time burst
    emit_amount [1,2]          #how many particles to emit per emit-trigger. bracket for a random range
    emit_direction [0, 1]  #x, y direction to emit particles in
    emit_spread 45         #angular spread for particle emission (360 for all directions)
    emit_box [.5, 0]       #initial location randomization box for particles (width, height)
    emit_offset [0,0]      #x, y offsets for where to emit particles from

    #particle animation properties
    particle_lifetime [.5,1]           #how many seconds a particle lives for (brackets for a random range)
    initial_speed [5,15]               #particle initial speed (bracket for a random range)
    initial_rotation [0,360]            #particle initial rotation (bracket for a random range) (ex: [0,360])
    rotation_speed [-180,180]              #rotating over time (bracket for random range)
    force [0, -40]                #x, y for a constant force (gravity or wind, etc)
    friction [1, 1]               #x, y friction parameters
    size_start [1,1.5]                  #size at start (brackets for a random range)
    size_end [.5, 0]                   #size at end (brackets for a random range)
    alpha_start [1,1]                  #size at start (brackets for a random range)
    alpha_end [1, 1]                   #size at end (brackets for a random range)
    face_moving_direction false       #false = off, true = on. rotate to match speed. initial_rotation becomes an offset if this is on
    speed_scale [0,0]             #scales x by speed (best used with face_moving_direction)
    inherit_speed [0,0]        #particles inherit the speed of their emitter (emitter speed is multiplied by this value)
    tags [collide_tile, collide_water, bounce_tile, collide_air] #tags exposed to the engine [collide_tile, collide_water, collide_air (for bubbles)]
}
